<script>
(function() {
    const iframe = document.getElementById('heart-viewer');
    const client = new Sketchfab(iframe);

    const affectedCodes = "{{ affected_code|default:'NORMAL' }}".split(',');
    const affectedArea = "{{ affected_area|default:'N/A' }}";

    const floatingLabel = document.getElementById('ai-floating-label');
    const labelText = document.getElementById('label-text');

    client.init('a3f0ea2030214a6bbaa97e7357eebd58', {
        success: function(api) {
            api.start();

            api.addEventListener('viewerready', function() {

                document.getElementById('sync-status').innerText = "Analyzing...";

                api.setHighlightOptions({
                    outlineColor: [1, 0, 0],
                    outlineWidth: 5,
                    fillColor: [1, 0, 0],
                    fillAlpha: 0.3
                });

                api.getNodeMap(function(err, nodes) {

                    let found = false;

                    Object.values(nodes).forEach(node => {
                        if(node.name && node.name.toLowerCase().includes(affectedCodes[0].toLowerCase())){
                            api.setHighlight(node.instanceID, 1);
                            found = true;
                        }
                    });

                    if(found){
                        labelText.innerText = affectedArea;
                        floatingLabel.style.display = 'block';
                        document.getElementById('sync-status').innerText = "Abnormal Detected";
                    } else {
                        document.getElementById('sync-status').innerText = "Normal";
                    }

                });

            });
        }
    });

})();
</script>
