def Test(request):
    # Debug: Print session info
    print(f"[DEBUG Test] Session key: {request.session.session_key}")
    print(f"[DEBUG Test] Student logged in: {request.session.get('student', False)}")
    
    if request.method == 'POST' and request.FILES.get('image'):
        uploaded_file = request.FILES['image']
        fs = FileSystemStorage()
        filename = fs.save(uploaded_file.name, uploaded_file)

        uploaded_file_url = fs.url(filename)
        full_image_path = fs.path(filename)

        model = load_model("model/model.h5")
        labels = np.load("model/labels.npy", allow_pickle=True).item()

        # Preprocess
        img = load_img(full_image_path, target_size=(48, 48))
        img_array = img_to_array(img) / 255.0
        img_array = np.expand_dims(img_array, axis=0)

        # Predict
        pred = model.predict(img_array)
        class_id = np.argmax(pred)
        predicted_label = labels[class_id]

        # Risk Logic
        if "Normal" in predicted_label:
            risk_level = "Low"
            bpm = 72
        elif "Mild" in predicted_label or "Maybe" in predicted_label:
            risk_level = "Medium"
            bpm = 95
        elif "History" in predicted_label:
            risk_level = "Medium"
            bpm = 68
        else:
            risk_level = "High"
            bpm = 130

        # ECG Signal (NumPy array)
        ecg_signal = np.sin(np.linspace(0, 20, 300))
        
        # MULTI-REGION DETECTION: Allow for multiple defect areas
        selected_regions = []
        
        # TOTAL ENTROPY: Use filename, image features, confidence, AND timestamp
        import hashlib, time
        img_avg = np.mean(img_array)
        img_std = np.std(img_array)
        confidence_seed = int(np.max(pred) * 100000)
        entropy_string = f"{filename}{img_avg}{img_std}{confidence_seed}{time.time()}"
        dynamic_seed = int(hashlib.md5(entropy_string.encode()).hexdigest(), 16)
        
        p_label_lower = predicted_label.lower()
        
        # FIX: Check for 'abnormal' FIRST because 'abnormal' contains the word 'normal'
        if "abnormal" in p_label_lower or "arrhythmia" in p_label_lower:
            arr_options = [
                ("SA Node (RA)", "RA"),
                ("AV Node (RA/RV)", "RA"),
                ("Bundle of His (Ventricles)", "LV"),
                ("Purkinje Fibers (LV)", "LV"),
                ("Bundle Branches (RV)", "RV")
            ]
            # Select 1 or 2 regions
            num_regions = 2 if (dynamic_seed % 10) > 6 else 1
            idx1 = dynamic_seed % len(arr_options)
            selected_regions.append(arr_options[idx1])
            if num_regions > 1:
                idx2 = (idx1 + 1) % len(arr_options)
                selected_regions.append(arr_options[idx2])
                
        elif "infarction" in p_label_lower or "mi" in p_label_lower:
            mi_options = [
                ("Anterior Wall (LV)", "LV"),
                ("Inferior Wall (RV)", "RV"),
                ("Lateral Wall (LV)", "LV"),
                ("Septal Wall (LV/RV)", "LV"),
                ("Posterior Wall (LV/RV)", "LV")
            ]
            num_regions = 2 if (dynamic_seed % 10) > 4 else 1
            idx1 = dynamic_seed % len(mi_options)
            selected_regions.append(mi_options[idx1])
            if num_regions > 1:
                idx2 = (idx1 + 1) % len(mi_options)
                selected_regions.append(mi_options[idx2])
                
        elif "normal" in p_label_lower:
            selected_regions.append(("No damaged regions - Healthy", "NORMAL"))
        else:
            # Fallback for other abnormalities
            selected_regions.append(("Diffuse Cardiac Irregularity", "LV,RV"))

        # Combine for DB storage (comma-separated codes, '&' separated areas)
        affected_area = " & ".join([r[0] for r in selected_regions])
        affected_code = ",".join(list(set([r[1] for r in selected_regions])))

        # Generate heart image (pass ECG signal for affected-area highlights)
        try:
            is_normal = (risk_level == "Low")
            heart_image = generate_heart_image(is_normal, ecg_signal.tolist(), context_text=predicted_label)
        except Exception as e:
            print(f"Error generating heart image: {e}")
            heart_image = None

        # Save record to DB (if model exists)
        try:
            userid = request.user if getattr(request, 'user', None) and request.user.is_authenticated else None
            confidence = float(np.max(pred)) if hasattr(pred, '__len__') else float(pred)
            record = ECGRecord.objects.create(
                uploaded_by=userid,
                filename=os.path.basename(full_image_path),
                result=str(predicted_label),
                confidence=confidence,
                risk_level=risk_level,
                bpm=bpm,
                affected_area=affected_area,
                affected_code=affected_code,
                file_type='image',
                heart_image=heart_image
            )
            print(f"[DEBUG] Saved ECG record to database: ID={record.id}, Result={predicted_label}, Risk={risk_level}, BPM={bpm}, Area={affected_area}")
        except Exception as e:
            record = None

        # store a short prediction history in session for student dashboard
        try:
            import datetime as _dt
            item = {
                'label': str(predicted_label),
                'risk_level': risk_level,
                'bpm': int(bpm),
                'time': _dt.datetime.now().strftime('%Y-%m-%d %H:%M'),
                'affected_area': str(affected_area),
                'affected_code': str(affected_code)
            }
            
            # Get existing history
            hist = request.session.get('pred_history', [])
            print(f"[DEBUG] Current history length: {len(hist)}")
            
            # Add new item at the beginning
            hist.insert(0, item)
            
            # Keep most recent 20
            request.session['pred_history'] = hist[:20]
            
            # Force session to save
            request.session.modified = True
            request.session.save()
            
            print(f"[DEBUG] Saved prediction to session. New history length: {len(request.session['pred_history'])}")
            print(f"[DEBUG] Saved item: {item}")
        except Exception as e:
            print(f"[ERROR] Error storing prediction history: {e}")
            import traceback
            traceback.print_exc()

        return render(request, 'UserApp/ImagePredction.html', {
            'data': predicted_label,
            'image': uploaded_file_url,
            'risk_level': risk_level,
            'ecg_signal': ecg_signal.tolist() , 
            'bpm': bpm,
            'heart_image': heart_image,
            'affected_area': affected_area,
            'affected_code': affected_code,
            'record': record
        })

    return render(request, 'UserApp/ImagePredction.html', {
        'data': 'Please upload an image.'
    })

# Gen 2

def generate_heart_image(is_normal=True):
    """Generate heart image using OpenAI or fallback"""
    try:
        api_key = os.environ.get("OPENAI_API_KEY")
        if not api_key:
            raise Exception("No API key")

        client = OpenAI(api_key=api_key)

        prompt = "healthy human heart" if is_normal else "damaged human heart"

        result = client.images.generate(
            model="dall-e-2",
            prompt=prompt,
            size="512x512"
        )

        img_data = base64.b64decode(result.data[0].b64_json)

        filename = f"heart_{uuid.uuid4().hex}.png"
        filepath = os.path.join(settings.MEDIA_ROOT, filename)

        with open(filepath, "wb") as f:
            f.write(img_data)

        return filename

    except:
        # fallback simple image
        filename = f"fallback_{uuid.uuid4().hex}.png"
        filepath = os.path.join(settings.MEDIA_ROOT, filename)

        from PIL import Image
        img = Image.new("RGB", (300, 300), (255, 0, 0 if not is_normal else 0))
        img.save(filepath)

        return filename

def HeartAnatomy(request):
    if not request.session.get("student"):
        return redirect("StudentLogin")
    
    # Priority 1: Current Session Activity (for instant feedback)
    hist = request.session.get('pred_history', [])
    latest_record = None
    
    if hist:
        latest = hist[0]
        # Create a mock-like object or dict for the context
        context = {
            'affected_code': latest.get('affected_code', 'NORMAL'),
            'affected_area': latest.get('affected_area', 'Healthy Trace'),
            'risk_level': latest.get('risk_level', 'Low'),
            'prediction': latest.get('label', 'Active Analysis')
        }
        print(f"[DEBUG Anatomy] Using session data: {context['affected_code']}")
    else:
        # Priority 2: Database Fallback
        latest_record = ECGRecord.objects.all().order_by('-id').first()
        context = {
            'affected_code': latest_record.affected_code if latest_record else 'NORMAL',
            'affected_area': latest_record.affected_area if latest_record else 'N/A',
            'risk_level': latest_record.risk_level if latest_record else 'Low',
            'prediction': latest_record.result if latest_record else 'No data loaded'
        }
        print(f"[DEBUG Anatomy] Using database fallback: {context['affected_code']}")
    
    return render(request, 'UserApp/HeartAnatomy.html', context)

        return HttpResponse(f'<h2 style="color:red;">Prediction record #{record_id} not found</h2><a href="/UserApp/Test">Back to Test</a>', status=404)

        print(f"[ERROR PredictionDetail] {str(e)}")
        from django.http import HttpResponse
        return HttpResponse(f'<h2 style="color:red;">Error loading prediction: {str(e)}</h2>', status=500)

            for record in records:
                result_text = record.result[:40] if record.result else "N/A"
                heart_img_text = str(record.heart_image) if record.heart_image else "<span style='color:red;'>EMPTY</span>"
                
                # Check if file exists
                file_exists = "✗ NO"
                file_size = 0
                if record.heart_image:
                    try:
                        full_path = record.heart_image.path
                        import os
                        if os.path.exists(full_path):
                            file_exists = "✓ YES"
                            file_size = os.path.getsize(full_path)
                        else:
                            file_exists = f"<span style='color:red;'>✗ NO (path: {full_path})</span>"
                    except Exception as e:
                        file_exists = f"<span style='color:red;'>ERROR: {str(e)[:30]}</span>"
                
                action_link = f'<a href="/media/{record.heart_image}" target="_blank">View</a>' if record.heart_image else "N/A"
                
                html += f'<tr>'
                html += f'<td>{record.id}</td>'
                html += f'<td>{result_text}</td>'
                html += f'<td>{record.risk_level}</td>'
                html += f'<td><small>{heart_img_text}</small></td>'
                html += f'<td>{file_exists} ({file_size} bytes)</td>'
                html += f'<td>{action_link}</td>'
                html += f'</tr>'
            
            html += '</table>'
        else:
            html += '<p>No records found.</p>'
        
        html += '<br><a href="/UserApp/index">Back to Home</a>'
        
        from django.http import HttpResponse
        return HttpResponse(html)
    except Exception as e:
        from django.http import HttpResponse
        return HttpResponse(f'<h2 style="color:red;">Error checking database: {str(e)}</h2><pre>{str(e)}</pre>')



def AdminAction(request):
    if request.method == "POST":
        username = request.POST.get('username')
        password = request.POST.get('password')

        print("USERNAME:", username)
        print("PASSWORD:", password)

        user = authenticate(request, username=username, password=password)

        print("USER OBJECT:", user)

        if user is not None:
            print("IS STAFF:", user.is_staff)

            if user.is_staff:
                login(request, user)
                print("LOGIN SUCCESS")
                return redirect('/admin/')
            else:
                return render(request, 'index.html', {'data': 'Not an admin'})
        else:
            return render(request, 'index.html', {'data': 'Invalid credentials'})

    return redirect('/')
