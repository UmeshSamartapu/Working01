<script>
(async function() {

    const viewer = document.getElementById('heart-viewer');

    const affectedCodes = "{{ affected_code|default:'NORMAL' }}".split(',');
    const affectedArea = "{{ affected_area|default:'N/A' }}";

    const floatingLabel = document.getElementById('ai-floating-label');
    const labelText = document.getElementById('label-text');
    const status = document.getElementById('sync-status');

    viewer.addEventListener('load', async () => {

        status.innerText = "Analyzing...";

        const model = viewer.model;
        if (!model) {
            status.innerText = "Model Load Failed";
            return;
        }

        const materials = model.materials;

        let normalMat = null;
        let targetMat = null;

        // ✅ DEBUG MATERIALS
        console.log("Available Materials:");
        materials.forEach(mat => {
            console.log(mat.name);
        });

        // ✅ SMART MATCHING
        materials.forEach(mat => {
            const name = mat.name.toLowerCase();

            if (name.includes("normal")) {
                normalMat = mat;
            }

            // fallback → first material
            if (!targetMat) {
                targetMat = mat;
            }
        });

        // If still null → assign safely
        if (!targetMat && materials.length > 0) {
            targetMat = materials[0];
        }

        // =============================
        // 🔥 LOGIC
        // =============================
        if (affectedCodes[0] !== "NORMAL") {

            status.innerText = "⚠ Abnormal Detected";

            labelText.innerText = affectedArea;
            floatingLabel.style.display = "block";

            // ✅ PULSE EFFECT
            let intensity = 0;

            setInterval(() => {
                intensity = (Math.sin(Date.now() / 200) + 1) / 2;

                if (targetMat) {
                    targetMat.setEmissiveFactor([
                        intensity, 0, 0
                    ]);
                }
            }, 50);

            // dim normal
            if (normalMat) {
                normalMat.setEmissiveFactor([0, 0, 0]);
            }

            // camera focus
            viewer.cameraOrbit = "0deg 75deg 2.2m";

        } else {

            status.innerText = "✅ Normal";
            floatingLabel.style.display = "none";

            if (targetMat) {
                targetMat.setEmissiveFactor([0, 0, 0]);
            }

            if (normalMat) {
                normalMat.setEmissiveFactor([0, 0, 0]);
            }
        }

    });

})();
</script>
